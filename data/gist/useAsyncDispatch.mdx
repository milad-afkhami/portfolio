title: "useAsyncDispatch"
category: "hook"
summary: "Async version of redux dispatch function"
---

## Source

```typescript
import { useDispatch } from "react-redux";

export default useAsyncDispatch;

const useAsyncDispatch = () => {
  const dispatch = useDispatch();

  return (action, payload) =>
    new Promise((resolve, reject) =>
      dispatch(action.trigger({ resolve, reject, ...payload }))
    );
};
```

You need to take care of settling this promise in your store.
It should be something like this:

```typescript
try {
  // ...
  action?.payload?.resolve?.(result);
} catch (error) {
  action?.payload?.reject?.(error);
}
```

## Usage

```typescript
const dispatch = useAsyncDispatch();

dispatch(profileActions, { userId: "USER_ID" })
  .then(() => {})
  .catch(() => {})
  .finally(() => {});
```
