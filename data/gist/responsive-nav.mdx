---
title: "Responsive Navbar"
category: "javascript"
summary: "Creating a responsive navbar without having to create and render multiple components and hide one of them on each screen."
---

See [here](https://github.com/milad-afkhami/portfolio/blob/main/components/Layout/Header/Nav/index.tsx) for a complete solution.

```jsx
import Div from "@kits/Div";
import dynamic from "next/dynamic";
import { useMediaQuery } from "react-responsive";
import breakpoints from "@stylesheets/constants/breakpoints.json";

const HeaderMenu = dynamic(() => import("./Desktop"), { ssr: false });
const HeaderResponsiveMenu = dynamic(() => import("./Responsive"), {
  ssr: false,
});

export default HeaderNav;
import type { FC } from "react";

interface HeaderNavProps {}

const HeaderNav: FC<HeaderNavProps> = (props) => {
  const isBigScreen = useMediaQuery({ minWidth: breakpoints.md });
  const isResponsiveScreen = useMediaQuery({ maxWidth: breakpoints.md });

  return (
    <Div display="flex">
      {isResponsiveScreen && <HeaderResponsiveMenu />}
      {isBigScreen && <HeaderMenu />}
    </Div>
  );
};
```
