### _renders and translates a custom span element with needed stylesheets._

## Usage

```jsx
<Text size="md-b" color="text-secondary" lineHeight="1.5" truncate>
  a.b.c
</Text>
```

## Source

```jsx
import React from "react";
import styled from "styled-components";
import { c, setTextTypography } from "@stylesheets";
import { withTranslation } from "react-i18next";

const Text = withTranslation()(
  styled.span.attrs(
    ({
      i18n,
      translationOptions = {},
      translationVariables = {},
      t,
      children,
      direction,
      tag,
      noTranslation,
      ...rest
    }) => {
      // const dir = direction || (i18n.language === "fa" ? "rtl" : "ltr");
      return {
        ...(tag ? { as: tag } : {}),
        children: noTranslation
          ? children
          : t(children, {
              ...translationOptions,
              ...translationVariables,
            }),
        ...rest,
      };
    }
  )(
    ({
      color,
      size,
      bold,
      m,
      p,
      css,
      truncate,
      width,
      mw,
      Mw,
      height,
      mh,
      Mh,
      align,
      hoverColor,
      zoomOnHover,
      lineHeight,
      ...restProps
    }) => ({
      ...(color ? { color: c(color) } : {}),
      ...setTextTypography(size, { bold }),
      margin: 0,
      transition: "var(--pace-fast)",
      ...(width ? { width } : {}),
      ...(mw ? { minWidth: mw } : {}),
      ...(Mw ? { maxWidth: Mw } : {}),
      ...(height ? { height } : {}),
      ...(mh ? { minHeight: mh } : {}),
      ...(Mh ? { maxHeight: Mh } : {}),
      ...(align ? { textAlign: align } : {}),
      ...(lineHeight ? { lineHeight } : {}),
      ...(css || {}),
      ...(truncate
        ? { textOverflow: "ellipsis", overflow: "hidden", whiteSpace: "nowrap" }
        : {}),
      ...(hoverColor
        ? {
            ":hover": {
              ...(hoverColor ? { color: c(hoverColor) } : {}),
              ...(zoomOnHover ? { transform: "scale(1.05)" } : {}),
            },
          }
        : {}),
      "-moz-font-feature-settings": '"ss02"',
      "-webkit-font-feature-settings": '"ss02"',
      "font-feature-settings": '"ss02"',
    })
  )
);
```
