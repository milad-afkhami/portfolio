### _A customizable useState hook to deal with binary states._

### Source

<Code
  language="javascript"
  code={`import { useState, useCallback, useDebugValue, useMemo } from "react";
function useDisclosure(name = "", defaultOpen: Boolean = false) {
  const [isOpen, setIsOpen] = useState(defaultOpen);
  useDebugValue({ ["is" + capitalName + "Open"]: isOpen });
  
  const open = useCallback(() => setIsOpen(true), []);
  const close = useCallback(() => setIsOpen(false), []);
  const toggle = useCallback(() => setIsOpen((state) => !state), []);
  
  const capitalName = useMemo(() => (name?.charAt?.(0)?.toUpperCase?.() || "") + (name?.slice?.(1)?.toLowerCase?.() || "");
  
  return {
    ["is" + capitalName + "Open"]: isOpen,
    ["open" + capitalName]: open,
    ["close" + capitalName]: close,
    ["toggle" + capitalName]: toggle,
    // aliases:
    [name ? name + "State" : "state"]: isOpen,
    ["show" + capitalName]: open,
    ["hide" + capitalName]: close,
  };
}
`}
/>
